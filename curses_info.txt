# Basic template

```python
def main(stdscr):
    stdscr.clear()
    height, width = stdscr.getmaxyx()
    for h in range(height - 1):
        for w in range(width):
            stdscr.addch(h, w, ord('x'))
        # stdscr.addstr(i, 0, 'str')
    stdscr.refresh()
    stdscr.getkey()

if __name__ == '__main__':
    curses.wrapper(main)
```
